## 4. 开关事件处理 (第 431-438 行)

```javascript
const web5Toggle = document.getElementById('wt-web5-toggle');
if (web5Toggle) {
  web5Toggle.onchange = (e) => {
    Storage.set(CONFIG.STORAGE_KEYS.WEB5_MODAL, e.target.checked ? '0' : '1');
    Toast.show('弹窗设置已更新，即将刷新', 'success');
    setTimeout(() => window.location.reload(), 800);
  };
}
// checked=true → 设置'0' → 开启弹窗
// checked=false → 设置'1' → 关闭弹窗
```

## 5. popup.html 中的对应开关 (new-features.js 第 274-342 行)

```javascript
export function initializeWeb5ModalToggle(tabId) {
  const toggle = document.getElementById('web5-modal-toggle');
  const MODAL_KEY = 'modalOff666';

  // 加载状态
  if (value === '1') {
    toggle.checked = false; // 关闭状态
  } else {
    toggle.checked = true; // 开启状态
  }

  // 监听开关
  if (isEnabled) {
    // 开启弹窗 = 设置为'0'
    await chrome.tabs.sendMessage(tabId, {
      type: 'SET_LOCAL_STORAGE',
      key: MODAL_KEY,
      value: '0'
    });
  } else {
    // 关闭弹窗 = 设置为'1'
    await chrome.tabs.sendMessage(tabId, {
      type: 'SET_LOCAL_STORAGE',
      key: MODAL_KEY,
      value: '1'
    });
  }
}
```
